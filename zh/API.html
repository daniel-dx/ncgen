      # API

      const { generate } = require("ncgen"); // or import { generate } from "ncgen"

      # generate(config, options) ⇒ promise

      Call ncgen through node api form

      Returns: promise - Promise
      Params

      • config string | object - Configuration file path or configuration object
      • options object - Options
        • .type string - "m" (main command) or "s" (subcommand)
        • [.cwd] string - Current working directory of the child process.
        • [.command] string - The name of the executed subcommand. Only needed when type is "s"
        • [.answers] object - Provided when you want to skip interactive questioning

      Example

      const { generate } = require("ncgen"); // or import { generate } from "ncgen"

// Execute the main command
generate('path/to/ncgen-config.js', { type: 'm' })
// or
const ncgenConfig = require('path/to/ncgen-config.js')
generate(ncgenConfig, { type: 'm' })

// Execute the main command with answer data
generate('path/to/ncgen-config.js', { type: 'm', answers: { projectName: 'demo', author: 'daniel' } })

// Execute the sub command
generate('path/to/ncgen-config.js', { type: 's', command: 'add-component' })

// Execute the sub command with answer data
generate('path/to/ncgen-config.js', { type: 's', command: 'add-component', answers: { category: 'busi', name: 'hello world' } })

      # api

      # transformStr(str) ⇒ object

      Convert input string into a variety of commonly used formats

      Returns: object - Strings in a variety of commonly used formats
      Params

      • str string - Input string.

      Example

      api.transformStr('demo-name')
api.transformStr('demoName')
api.transformStr('DemoName')
api.transformStr('Demo Name')
api.transformStr('Demo name')
api.transformStr('demo_name')

// returns:
{
  "kebabCase": "demo-name",
  "camelCase": "demoName",
  "snakeCase": "demo_name",
  "upperFirstCamelCase": "DemoName",
  "title": "Demo Name",
  "humanized": "Demo name"
}

      # replace(content, rules) ⇒ string

      Replacement method that can process multiple replacement rules at the same time

      Returns: string - Replaced content
      Params

      • content string - What's replaced
      • rules object | array - Multiple replacement rules. Object format: {"regular expression": "replace to"]}; Array format: [[/regular expression/, 'replace to']]

      Example

      const content = "ncgen is a very niice coded generator";
api.replace(content, {
  "\\sniice\\s": " nice ",
  coded: "code",
});
api.replace(content, [
  [/\sniice\s/, " nice "],
  ["coded", "code"],
]);

// returns
ncgen is a very nice code generator

      # insertBefore(content, rules) ⇒ string

      Batch processing: insert the specified string into the previous line of the specified matching string Note: only string matching, does not support regular expressions

      Returns: string - The content containing the inserted string
      Params

      • content string - Content being processed
      • rules object - Insert rules. Data format: {"matched string": "inserted string"}

      Example

      const content = "a\nc\ne";
const result = api.insertBefore(content, {
  c: "b",
  e: "d",
});

// returns
a
b
c
d
e

      # insertAfter(content, rules) ⇒ string

      Batch processing: insert the specified string into the next line of the specified matching string Note: only string matching, does not support regular expressions

      Returns: string - The content containing the inserted string
      Params

      • content string - Content being processed
      • rules object - Insert rules. Data format: {"matched string": "inserted string"}

      Example

      const content = "a\nc\ne";
const result = api.insertAfter(content, {
  a: "b",
  c: "d",
});

// returns
a
b
c
d
e

      # listDirs(dirPath, [excludes]) ⇒ Array.<string>

      List the directory name of the specified path

      Returns: Array.<string> - directory name list
      Params

      • dirPath string - Specify the path
      • [excludes] Array.<string> | function - Excluded directory name

      Example

      // Assuming your project directory structure is
.
├── bin
├── node_modules
├── src
├── test
└── package.json

api.listDirs("./");
// returns
["bin", "node_modules", "src", "test"]

api.listDirs("./", ["bin", "node.*"]);
// returns
["src", "test"]

api.listDirs(
  "./",
  (dir) => ["node_modules", "bin"].indexOf(dir) >= 0
);
// returns
["src", "test"]

      # listFiles(dirPath, [excludes]) ⇒ Array.<string>

      List the file name of the specified path

      Returns: Array.<string> - file name list
      Params

      • dirPath string - Specify the path
      • [excludes] Array.<string> | function - Excluded file name

      Example

      // Assuming your project directory structure is
.
├── bin
├── node_modules
├── src
├── jest.config.js
├── README.md
├── package-lock.json
└── package.json

api.listFiles("./");
// returns
["jest.config.js", "README.md", "package-lock.json", "package.json"]

api.listFiles("./", ["package-lock.*", "README.md"]);
// returns
["jest.config.js", "package.json"]

api.listFiles(
  "./",
  (file) => ["package-lock.json", "README.md"].indexOf(file) >= 0
);
// returns
["jest.config.js", "package.json"]
